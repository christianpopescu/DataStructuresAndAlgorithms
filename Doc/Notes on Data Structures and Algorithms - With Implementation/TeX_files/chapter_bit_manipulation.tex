\section{Some interesting functions and examples}

\begin{itemize}
	\item set the k\textsuperscript{th} bit of x to 1. 
	\begin{lstlisting}
	\x|(1<<k)
	\end{lstlisting}
	\item set the k\textsuperscript{th} bit of x to 0.
	\begin{lstlisting}
	x&~(1<<k)
	\end{lstlisting}
	\item inverts the k\textsuperscript{th} bit of x to 0.
	\begin{lstlisting}
	x^(1<<k)
	\end{lstlisting}

\end{itemize}
